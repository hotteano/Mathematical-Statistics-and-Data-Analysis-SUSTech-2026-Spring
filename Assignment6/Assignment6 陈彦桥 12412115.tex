\section{Page 69, Problem 56}

For $Y = |X|$, $X \sim N(0, \sigma^2)$, when $y < 0$, we have $f_Y(y) = 0$ since $Y$ is non-negative. For $y \geq 0$, we have the inverse functions $x_1 = y$ and $x_2 = -y$. The derivative of both inverse functions in absolute value is $1$. 
\begin{align*}
f_Y(y) &= f_X(y)\left|\frac{dx_1}{dy}\right| + f_X(-y)\left|\frac{dx_2}{dy}\right| \\
&= f_X(y) + f_X(-y) \\ 
&= \frac{1}{\sqrt{2\pi}\sigma} \exp\left(-\frac{y^2}{2\sigma^2}\right) + \frac{1}{\sqrt{2\pi}\sigma} \exp\left(-\frac{(-y)^2}{2\sigma^2}\right) \\
&= \frac{2}{\sqrt{2\pi}\sigma} \exp\left(-\frac{y^2}{2\sigma^2}\right)
\end{align*}

Then the density function is 
$$
f_Y(y) = \begin{cases}
0, & y < 0 \\
\frac{2}{\sqrt{2\pi}\sigma} \exp\left(-\frac{y^2}{2\sigma^2}\right), & y \geq 0
\end{cases}
$$

\section{Page 69, Problem 59}

From the problem, $U \sim U(-1, 1)$, then $f_U(u) = \frac{1}{2}$ for $u \in [-1, 1]$. For $Y = U^2$, since $U \in [-1, 1]$, $Y \in [0, 1]$.
For $y \notin [0, 1]$, $f_Y(y) = 0$. 
For $y \in [0, 1]$, we have inverse functions $u_1 = \sqrt{y}$ and $u_2 = -\sqrt{y}$, with derivatives $\frac{du_1}{dy} = \frac{1}{2\sqrt{y}}$ and $\frac{du_2}{dy} = -\frac{1}{2\sqrt{y}}$.
\begin{align*}
f_Y(y) &= f_U(\sqrt{y})\left|\frac{1}{2\sqrt{y}}\right| + f_U(-\sqrt{y})\left|-\frac{1}{2\sqrt{y}}\right| \\
&= \frac{1}{2} \cdot \frac{1}{2\sqrt{y}} + \frac{1}{2} \cdot \frac{1}{2\sqrt{y}} \\
&= \frac{1}{2\sqrt{y}}
\end{align*}

Then the density function is
$$
f_Y(y) = \begin{cases}
\frac{1}{2\sqrt{y}}, & y \in (0, 1] \\
0, & \text{otherwise}
\end{cases}
$$

\section{Page 69, Problem 62}

\begin{proof} 

From the definition of the cumulative distribution function (CDF) for $Y = aX + b$, we have
$$
F_Y(y) = P(Y \leq y) = P(aX + b \leq y)
$$
If $a > 0$, then we have
$$
F_Y(y) = P\left(X \leq \frac{y - b}{a}\right) = F_X\left(\frac{y - b}{a}\right)
$$
Taking the derivative with respect to $y$, we get:
$$
f_Y(y) = \frac{d}{dy}F_Y(y) = f_X\left(\frac{y - b}{a}\right) \cdot \frac{1}{a}
$$
If $a < 0$, then we have
$$
F_Y(y) = P\left(X \geq \frac{y - b}{a}\right) = 1 - P\left(X < \frac{y - b}{a}\right) = 1 - F_X\left(\frac{y - b}{a}\right)
$$
Taking the derivative with respect to $y$, we get:
$$
f_Y(y) = \frac{d}{dy}F_Y(y) = -f_X\left(\frac{y - b}{a}\right) \cdot \frac{1}{a}
$$

In either case ($a \neq 0$), we can write $f_Y(y) = \frac{1}{|a|} f_X\left(\frac{y - b}{a}\right)$, which completes the proof.
\end{proof}

\section{Supplementary Problem 2}

Notice that $\sin(x)$ increase on $[0, \frac{\pi}{2}]$, decrease on $[\frac{\pi}{2}, \pi]$ and $\sin(0) = \sin(\pi) = 0$, $\sin(\frac{\pi}{2}) = 1$. Thus for $Y = \sin(X)$, we have 
the inverse function $g_1^{-1}(y) = \arcsin(y)$ and $g_2^{-1}(y) = \pi - \arcsin(y)$, and the absolute value of the derivative of the inverse functions are $|(g_1^{-1})'(y)| = \frac{1}{\sqrt{1-y^2}}$ and $\left|(g_2^{-1})'(y)\right| = \left|-\frac{1}{\sqrt{1-y^2}}\right| = \frac{1}{\sqrt{1-y^2}}$. Then for $0 < y < 1$, we have

$$
f_Y(y) = \sum_{i=1}^2 \left|(g_i^{-1})'(y)\right| f_X(g_i^{-1}(y)) = \frac{1}{\sqrt{1-y^2}} f_X(\arcsin(y)) + \frac{1}{\sqrt{1-y^2}} f_X(\pi - \arcsin(y))
$$

From the given function, we have 
$$
f_Y(y) = \begin{cases}
0, & y \notin (0, 1) \\
\frac{1}{\sqrt{1-y^2}} \frac{2}{\pi^2} \arcsin(y) + \frac{1}{\sqrt{1-y^2}} \frac{2}{\pi^2} (\pi - \arcsin(y)), & y \in (0, 1)
\end{cases}
$$

To be concise, we have
$$
f_Y(y) = \begin{cases}
0, & y \notin (0, 1) \\
\frac{2}{\pi \sqrt{1-y^2}}, & y \in (0, 1)
\end{cases}
$$

\section{Supplementary Problem 4}

From the problem, $X \sim U(1,2)$. $Y = e^{2X}$ increase on $[1, 2]$, thus the inverse function is $g^{-1}(y) = \frac{1}{2} \ln(y), y \in [e^2, e^4]$ and the derivative of the inverse function is $(g^{-1})'(y) = \frac{1}{2y}$. Then for $e^2 \leq y \leq e^4$, we have
\begin{align*}
f_Y(y) &= (g^{-1})'(y) f_X(g^{-1}(y)) \\
&= \frac{1}{2y} \cdot 1 \\
&= \frac{1}{2y}
\end{align*}

Thus the density function is
$$
f_Y(y) = \begin{cases}
\frac{1}{2y}, & y \in [e^2, e^4] \\
0, & \text{otherwise}
\end{cases}
$$
